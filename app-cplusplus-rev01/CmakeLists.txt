cmake_minimum_required(VERSION 3.10)

# Set the project name and version
project(MyLoadProcessingApp VERSION 1.0)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Find the Qt package (make sure you have installed Qt)
find_package(Qt5 COMPONENTS Widgets REQUIRED)

# Add the executable target
add_executable(load_processor process_load.cpp)

# Link Qt libraries to the executable
target_link_libraries(load_processor Qt5::Widgets)
